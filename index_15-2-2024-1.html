Así, ¿podría existir una escritura efímera? Si esto fuera posible ni siquiera este experimento seria posible. Pero ¿podríamos imaginar un mundo donde la escritura deje de ser esclava del registro en piedra, en papel o en un byte?

packages = ["matplotlib", "networkx", "ssl", "pandas"] import io import matplotlib.pyplot as plt import networkx as nx import base64 def process_text(text): tokens = text.split() graph = nx.Graph() for i in range(len(tokens) - 1): graph.add_edge(tokens[i], tokens[i + 1]) # Ajusta el tamaño del gráfico aquí fig, ax = plt.subplots(figsize=(14, 8)) # Puedes ajustar estos valores según necesites nx.draw(graph, ax=ax, with_labels=True) plt.axis('off') # Esto quitará los ejes fig, ax = plt.subplots(figsize=(17, 20)) # Ajusta estos valores según las dimensiones del contenedor nx.draw(graph, ax=ax, with_labels=True) plt.axis('off') img = io.BytesIO() plt.savefig(img, format='png', bbox_inches='tight', pad_inches=0) # Reducir el espacio alrededor del gráfico img.seek(0) img_base64 = base64.b64encode(img.read()).decode('utf-8')
